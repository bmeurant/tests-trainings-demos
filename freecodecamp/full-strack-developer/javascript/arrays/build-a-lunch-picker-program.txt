const lunches = [];

function addLunchToEnd(lunchProg, lunch) {
  lunchProg.push(lunch);
  console.log(`${lunch} added to the end of the lunch menu.`);
  return lunchProg;
}

function addLunchToStart(lunchProg, lunch) {
  lunchProg.unshift(lunch);
  console.log(`${lunch} added to the start of the lunch menu.`);
  return lunchProg;
}

function removeLastLunch(lunchProg) {
  if (lunchProg.length === 0) {
    console.log("No lunches to remove.");
  } else {
    const lunch = lunchProg.pop();
    console.log(`${lunch} removed from the end of the lunch menu.`);
  }
  return lunchProg;
}

function removeFirstLunch(lunchProg) {
  if (lunchProg.length === 0) {
    console.log("No lunches to remove.");
  } else {
    const lunch = lunchProg.shift();
    console.log(`${lunch} removed from the start of the lunch menu.`);
  }
  return lunchProg;
}

function getRandomLunch(lunchProg) {
  if (lunchProg.length === 0) {
    console.log("No lunches available.");
  } else {
    const randomIdx = Math.floor(Math.random() * lunchProg.length);
    const lunch = lunchProg[randomIdx];
    console.log(`Randomly selected lunch: ${lunch}`);
  }
}

function showLunchMenu(lunchProg) {
  if (lunchProg.length === 0) {
    console.log("The menu is empty.");
  } else {
    console.log(`Menu items: ${lunchProg.join(", ")}`);
  }
}
