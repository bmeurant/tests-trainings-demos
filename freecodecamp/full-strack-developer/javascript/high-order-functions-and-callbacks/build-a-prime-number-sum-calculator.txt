function isPrime(n) {
  if (n <= 1) return false;
  for (let i = 2; i <= Math.sqrt(n); i++) {
    if (n % i === 0) return false;
  }
  return true;
};

function sumPrimes(num) {
  if (2 > num) return 0;
  const candidates = Array.from({ length: num + 1 }, (_, i) => i);
  return candidates.filter(isPrime).reduce((sum, n) => sum + n, 0);
}
