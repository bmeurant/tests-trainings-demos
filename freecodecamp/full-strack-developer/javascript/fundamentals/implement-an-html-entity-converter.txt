const htmlEntitiesMap = {
  '&': '&amp;',
  '<': '&lt;',
  '>': '&gt;',
  '"': '&quot;',
  "'": '&apos;'
};

function convertHTML(str) {
  const regex = /[&<>"']/g;

  return str.replace(regex, (match) => {
    return htmlEntitiesMap[match];
  });
}
